void solve()
{
    ll n;
    cin>> n;
    ll t = n;
    ll sum = 1;
    for(ll i = 0;i < vc.size() and i*i <= n;i++)
    {
        if(n % vc[i] == 0)
        {
            ll cnt = 1;
            while(n % vc[i] == 0)
                cnt++,n /= vc[i];
            sum *= (pow(vc[i],cnt)-1)/(vc[i]-1);
        }
    }
    if(n > 1)
        sum *= n+1;
    cout << sum-t << endl;
}
