void solve()
{
  /*
    1. Think Greedy
    2. Think Brute Force
    3. Think solution in reverse order
    4. Think DP [ check constraints carefully ]
    5. Check base cases for DP and prove solution for Greedy
    6. Think Graph
  */ 
    string s;
    cin >> s;
    map<char,int>mp;
    vector <string > vc = { "blue", "green", "yellow", "red", "purple", "orange", "pink", "grey", "cyan", "brown", "ash", "silver", "gold", "white", "black" };
    for(int i = 0;i < s.size();i++)
        mp[s[i]]++;
    int cnt = 0;
    int n = vc.size();
    int ans = 0;
    for(int mask=1;mask<(1<<n);mask++)
    {
        map<char,int>m;
        m = mp;
        int f = 1;
        for(int i=0;i<n;i++)
        {
            if(mask & (1<<i)){
                string ss = vc[i];
                for(int j = 0;j < ss.size();j++)
                {
                    if(m[ss[j]])
                        m[ss[j]]--;
                    else
                    {
                        f = 0;
                        break;
                    }
                }
            }
        }
        if(f)
            ans = max(ans,(ll)__builtin_popcount(mask));          
    }
    cout << ans << endl;
}
