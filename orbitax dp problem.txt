void solve()
{
    /*
        https://github.com/TamimEhsan/interview-questions-bangladesh/blob/master/docs%2Forbitax.md
        
        You are given a string S of length N and an integer K. Your task is to find the total number of subsequences in S that match the pattern P = "orbitaxian", where the difference in position between every consecutive character in the subsequence is no more than K.

        More formally, let the subsequence of P be represented by an array pos, where (1 ≤ pos[i] ≤ N) and S[pos[i]] = P[i]. Then for each i > 1, the condition pos[i] − pos[i-1] ≤ K must hold.

        The result should be returned modulo 10^9+7

        input:
        5
        11 1
        orbittaxian
        11 2
        orbittaxian
        20 1
        oorrbbiittaaxxiiaann
        20 2
        oorrbbiittaaxxiiaann
        20 6
        ossrbbijttaayxiibamn

        output:
        0
        2
        0
        11
        16
    */
    int n,k;
    cin>> n >> k;
    string s;
    cin>>s;
    string ss = "orbitaxIAn";
    map<char,deque<pair<int,int>>> mp;//char = {position,value}
    map<char,int>ans;//char = value;
    auto cal = [&](char prev,char cur,int ind)->void
    {   
        while(mp[prev].size() and ind-mp[prev].front().first > k)
        {
            ans[prev] -= mp[prev].front().second;
            ans[prev] = mo(ans[prev],mod);
            mp[prev].pop_front();
        }
        if(mp[prev].size())
        {
            ans[cur] += ans[prev];
            ans[cur] %= mod;
            mp[cur].push_back(mp(ind,ans[prev]));
        }
    };
    for(int i = 0;i < n;i++)
    {
        if(s[i] == 'o')
        {
            mp['o'].pb(mp(i,1));
            ans['o']++;
        }
        else if(s[i] == 'r')
            cal('o','r',i);
        else if(s[i] == 'b')
            cal('r','b',i);
        else if(s[i] == 'i')
            cal('b','i',i),cal('x','I',i);
        else if(s[i] == 't')
            cal('i','t',i);
        else if(s[i] == 'a')
            cal('t','a',i),cal('I','A',i);
        else if(s[i] == 'x')
            cal('a','x',i);
        else if(s[i] == 'n')
            cal('A','n',i);
    }
    cout << ans['n'] << endl;
}
